# 复制指令 v-copy

用于信息复制

需要引入 clipboard.js

<script src="//unpkg.com/clipboard"></script>

# 基础用法

复制
<template>
  <el-button v-copy="link" icon="el-icon-copy-document">复制</el-button>
</template>
<script>
export default {
  data() {
    return {
      link: "https://kwokronny.gitee.io/element-ui-saas-extend",
    };
  },
};
</script>
